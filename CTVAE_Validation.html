X = np.zeros((N, 3))

for i,d in enumerate(dists):
    X[start:(start+interval),0] = np.random.normal(d[0][0], d[0][1], int(N/3))

Y[(int(N/3) * 2):(int(N/3) * 3)] = 2
Y = np.array(pd.get_dummies(Y,drop_first=True))

def sample_from_decoder(tvae, decoder_path,transformer, n = 500, label=[0,0], cuda=False):
    # zero_prop : 疑似データにおけるtarget=0の割合, 出力にはtargetは含まないが、
    
    
    device = 'cuda' if torch.cuda.is_available() else 'cpu'
    decoder = torch.load(decoder_path, map_location=torch.device(device))
    #TVAE.decoder =decoder
    
    data_dim = n
    
    decoder.eval()
    
    label_ = torch.zeros((n,2)).to(device)
    label_[:, 0] = label[0]
    label_[:, 1] = label[1] #[0, 0] # target 1 , [1, 0] # target 2, [1, 1], target 3 
    
    steps = n // tvae.batch_size
    if n % tvae.batch_size != 0:
        steps += 1
    data = []
    for i in range(steps):
        if i != steps -1:
            batch_size = tvae.batch_size
        elif n % tvae.batch_size ==0:
            batch_size = tvae.batch_size
        else:
            batch_size = n % tvae.batch_size
            
        mean = torch.zeros(batch_size, tvae.embedding_dim)
        std = mean + 1
        label_batch = label_[(tvae.batch_size * i):min(tvae.batch_size * (i+1), n)  ]
        noise = torch.normal(mean=mean, std=std).to(device)
        #print(noise.shape)
        #print(label_batch.shape)
        fake, sigmas = decoder(noise,label_batch)
        fake = torch.tanh(fake)
        data.append(fake.detach().cpu().numpy())

    data = np.concatenate(data, axis=0)
    data = data[:n]
    return transformer.inverse_transform(data, sigmas.detach().cpu().numpy())

import matplotlib.pyplot as plt
import japanize_matplotlib
plt.style.use('seaborn')

def plothist(s=0, e=9,dfd=None, samplesd=None,total=None, cols=None):
    plt.figure(figsize=(15,15))
    for en,i in enumerate(range(s, e)):
        col = cols[i]
        plt.subplot(5,5, en+1)
        bot = np.min(total[col])
        up = np.max(total[col])
        bins = np.linspace(int(bot), int(up), 50)
        plt.hist(dfd[col], bins, alpha=0.5, edgecolor='black')
        plt.hist(samplesd[col],bins,alpha=0.5, edgecolor='black')
        plt.hist(total[col],bins,alpha=0.3, edgecolor='black', color='gray')
        plt.legend(['Original_Cond', 'Sampled','Original_Total'])
        plt.title(col, fontsize=15, fontname="MS Gothic")

    plt.tight_layout()
    plt.show()
